Analytics: Per-Project Operations Dashboard
What this step does
The Analytics page currently shows generic simulated data with no project filter. This step replaces it with a real operations dashboard that reads live data already present in the Naya database — task completion rates, content pipeline status, campaign performance, and the current week's strategy — all filtered per project. No new schema migrations are needed. No fake data. Everything is computed from existing tables.

Task 1 — Server: add per-project analytics computation route
Add this route in server/routes.ts:
GET /api/analytics/project-summary?projectId=X
This route computes everything the dashboard needs in a single call. Logic:
tsapp.get('/api/analytics/project-summary', isAuthenticated, async (req: any, res) => {
  const userId = req.user.claims.sub;
  const projectId = req.query.projectId ? parseInt(req.query.projectId as string) : null;

  // 1. Tasks: all tasks for this user+project
  const allTasks = await storage.getTasks(userId, projectId ?? undefined);
  const totalTasks = allTasks.length;
  const completedTasks = allTasks.filter(t => t.completed).length;
  const pendingTasks = totalTasks - completedTasks;
  const completionRate = totalTasks > 0 ? Math.round((completedTasks / totalTasks) * 100) : 0;

  // Tasks by energy type
  const tasksByEnergy = allTasks.reduce((acc: Record<string, number>, t) => {
    const key = t.taskEnergyType || 'other';
    acc[key] = (acc[key] || 0) + 1;
    return acc;
  }, {});

  // Tasks by type
  const tasksByType = allTasks.reduce((acc: Record<string, number>, t) => {
    acc[t.type] = (acc[t.type] || 0) + 1;
    return acc;
  }, {});

  // 2. Content pipeline — counts by contentStatus
  const allContent = await storage.getContent(userId, 500, projectId ?? undefined);
  const contentByStatus = { idea: 0, draft: 0, ready: 0, published: 0 };
  for (const c of allContent) {
    const s = (c.contentStatus || 'idea') as keyof typeof contentByStatus;
    if (s in contentByStatus) contentByStatus[s]++;
  }
  const contentByPlatform = allContent.reduce((acc: Record<string, number>, c) => {
    acc[c.platform] = (acc[c.platform] || 0) + 1;
    return acc;
  }, {});

  // 3. Campaigns
  const allCampaigns = await storage.getCampaigns(userId, projectId ?? undefined);
  const activeCampaigns = allCampaigns.filter(c => c.status === 'active').length;
  const completedCampaigns = allCampaigns.filter(c => c.status === 'completed').length;
  const draftCampaigns = allCampaigns.filter(c => c.status === 'draft').length;
  const totalCampaignTasks = allCampaigns.reduce((sum, c) => {
    return sum + ((c.generatedTasks as any[])?.length || 0);
  }, 0);

  // 4. Current week strategy report
  const now = new Date();
  const currentWeek = now.getFullYear() + '-' + String(now.getMonth() + 1).padStart(2, '0') + '-W' + Math.ceil(now.getDate() / 7);
  const strategyReport = projectId
    ? await storage.getStrategyReport(userId, currentWeek, projectId)
    : await storage.getStrategyReport(userId, currentWeek);

  // 5. Weekly task completion trend (last 4 weeks, by scheduledDate prefix)
  // Group completed tasks by week label — approximate from scheduledDate
  const weeklyCompletion: Array<{ week: string; completed: number; total: number }> = [];
  for (let i = 3; i >= 0; i--) {
    const d = new Date();
    d.setDate(d.getDate() - i * 7);
    const weekLabel = `W-${i === 0 ? 'now' : i}`;
    const weekStart = new Date(d);
    weekStart.setDate(weekStart.getDate() - weekStart.getDay());
    const weekEnd = new Date(weekStart);
    weekEnd.setDate(weekEnd.getDate() + 6);
    const weekStartStr = weekStart.toISOString().slice(0, 10);
    const weekEndStr = weekEnd.toISOString().slice(0, 10);
    const weekTasks = allTasks.filter(t => t.scheduledDate && t.scheduledDate >= weekStartStr && t.scheduledDate <= weekEndStr);
    weeklyCompletion.push({
      week: weekLabel,
      completed: weekTasks.filter(t => t.completed).length,
      total: weekTasks.length,
    });
  }

  res.json({
    tasks: { total: totalTasks, completed: completedTasks, pending: pendingTasks, completionRate, byEnergy: tasksByEnergy, byType: tasksByType },
    content: { total: allContent.length, byStatus: contentByStatus, byPlatform: contentByPlatform },
    campaigns: { active: activeCampaigns, completed: completedCampaigns, draft: draftCampaigns, totalTasksGenerated: totalCampaignTasks },
    strategy: strategyReport ? { focus: strategyReport.focus, recommendations: strategyReport.recommendations, weeklyPlan: strategyReport.weeklyPlan } : null,
    weeklyCompletion,
  });
});
