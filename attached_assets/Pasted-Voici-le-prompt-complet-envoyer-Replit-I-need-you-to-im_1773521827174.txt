In the implementation:
```ts
async deleteCampaignFutureTasks(campaignId: number, fromDate: string): Promise<number> {
  const deleted = await db.delete(tasks).where(
    and(
      eq(tasks.campaignId, campaignId),
      eq(tasks.completed, false),
      gte(tasks.scheduledDate, fromDate)
    )
  ).returning({ id: tasks.id });
  return deleted.length;
}
```

**4. `server/routes.ts` — add two new routes before `const httpServer = createServer(app)`:**

```ts
// Pause a campaign: save note + delete future incomplete tasks
app.post('/api/campaigns/:id/pause', isAuthenticated, async (req: any, res) => {
  try {
    const userId = req.user.claims.sub;
    const id = parseInt(req.params.id);
    const { pauseNote } = req.body as { pauseNote?: string };
    const campaign = await storage.getCampaign(id, userId);
    if (!campaign) return res.status(404).json({ message: "Campaign not found" });
    if (campaign.status !== 'active') return res.status(400).json({ message: "Campaign is not active" });
    const today = new Date().toISOString().slice(0, 10);
    const deleted = await storage.deleteCampaignFutureTasks(id, today);
    const updated = await storage.updateCampaign(id, userId, {
      status: 'paused',
      pauseNote: pauseNote?.trim() || null,
    });
    res.json({ campaign: updated, tasksRemoved: deleted });
  } catch (error) {
    console.error("Error pausing campaign:", error);
    res.status(500).json({ message: "Failed to pause campaign" });
  }
});

// Resume a paused campaign: optionally apply AI note, then re-launch
app.post('/api/campaigns/:id/resume', isAuthenticated, async (req: any, res) => {
  try {
    const userId = req.user.claims.sub;
    const id = parseInt(req.params.id);
    const campaign = await storage.getCampaign(id, userId);
    if (!campaign) return res.status(404).json({ message: "Campaign not found" });
    if (campaign.status !== 'paused') return res.status(400).json({ message: "Campaign is not paused" });

    if (campaign.pauseNote) {
      const OpenAI = (await import('openai')).default;
      const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
      const existingTasks = (campaign.generatedTasks || []) as Array<any>;
      const prompt = `You are a campaign planning assistant. The user has paused a marketing campaign and left the following modification note:\n\n"${campaign.pauseNote}"\n\nHere are the current campaign tasks (JSON array):\n${JSON.stringify(existingTasks, null, 2)}\n\nPlease return a modified JSON array of tasks reflecting the user's requested changes. Keep the same structure (title, description, type, category, priority, estimatedDuration, taskEnergyType, phase). Return ONLY valid JSON, no explanation.`;
      const completion = await openai.chat.completions.create({
        model: 'gpt-4o',
        messages: [{ role: 'user', content: prompt }],
        response_format: { type: 'json_object' },
        temperature: 0.3,
      });
      const raw = completion.choices[0]?.message?.content || '{}';
      let parsed: any;
      try { parsed = JSON.parse(raw); } catch { parsed = {}; }
      const updatedTasks: any[] = Array.isArray(parsed) ? parsed : (parsed.tasks ?? existingTasks);
      await storage.updateCampaign(id, userId, { generatedTasks: updatedTasks });
    }

    await storage.updateCampaign(id, userId, { status: 'active', pauseNote: null });
    const resumeStartDate = new Date().toISOString().slice(0, 10);
    res.json({ message: "Campaign resumed", resumeStartDate });
  } catch (error) {
    console.error("Error resuming campaign:", error);
    res.status(500).json({ message: "Failed to resume campaign" });
  }
});
```

**5c.** Add these two mutations (after `deleteMutation`):
```ts
const pauseMutation = useMutation({
  mutationFn: async ({ id, note }: { id: number; note: string }) => {
    const res = await apiRequest("POST", `/api/campaigns/${id}/pause`, { pauseNote: note });
    return res.json();
  },
  onSuccess: (data: { campaign: Campaign; tasksRemoved: number }) => {
    queryClient.invalidateQueries({ queryKey: ["/api/campaigns", selectedProjectId] });
    queryClient.invalidateQueries({ queryKey: ["/api/tasks"] });
    setShowPauseDialog(false);
    setPauseNoteInput("");
    toast({ title: "Campaign paused", description: `${data.tasksRemoved} future task(s) removed from planning.` });
  },
  onError: () => toast({ title: "Failed to pause campaign", variant: "destructive" }),
});

const resumeMutation = useMutation({
  mutationFn: async ({ id, note }: { id: number; note: string }) => {
    if (note.trim()) {
      await apiRequest("PATCH", `/api/campaigns/${id}`, { pauseNote: note.trim() });
    }
    const resumeRes = await apiRequest("POST", `/api/campaigns/${id}/resume`, {});
    const resumeData = await resumeRes.json();
    const launchRes = await apiRequest("POST", `/api/campaigns/${id}/launch`, { startDate: resumeData.resumeStartDate });
    return launchRes.json();
  },
  onSuccess: (data: { campaign: Campaign; tasksCreated: number }) => {
    queryClient.invalidateQueries({ queryKey: ["/api/campaigns", selectedProjectId] });
    queryClient.invalidateQueries({ queryKey: ["/api/tasks"] });
    setResumeNoteInput("");
    toast({ title: "Campaign resumed 🚀", description: `${data.tasksCreated} task(s) rescheduled from today.` });
  },
  onError: () => toast({ title: "Failed to resume campaign", variant: "destructive" }),
});
```
