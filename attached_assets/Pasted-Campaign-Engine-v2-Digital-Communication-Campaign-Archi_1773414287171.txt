Campaign Engine v2 — Digital Communication Campaign Architecture
The problem with v1
The current campaign engine generates a flat task list. A real digital communication campaign is an orchestrated system with phases, narrative arcs, channels, content formats, messaging frameworks, and multi-week deployment logic. This step rebuilds the AI generation layer and the schema to reflect that, while keeping the existing UI structure and adding the necessary new display sections.

New interfaces:
export interface CampaignPhase {
  number: number;
  name: string;                    // e.g. "Foundation & Awareness"
  duration: string;                // e.g. "Weeks 1–2"
  objective: string;               // What this phase achieves
  keyActions: string[];            // 3-5 concrete actions
  successSignal: string;           // How you know this phase worked
}

export interface ChannelConfig {
  platform: string;               // linkedin | instagram | email | blog | podcast | etc.
  role: string;                   // Primary | Secondary | Amplification
  frequency: string;              // e.g. "3x/week", "1x/week"
  contentFormat: string[];        // e.g. ["carousel", "short-form video", "text post"]
  tone: string;                   // How the brand voice adapts on this channel
}

export interface ContentPiece {
  phase: number;                   // Which phase this belongs to
  week: string;                    // e.g. "Week 1", "Month 2"
  platform: string;
  format: string;                  // carousel | video | article | email | story | etc.
  angle: string;                   // The specific angle/hook for this piece
  pillar: string;                  // Content pillar it serves
  goal: string;                    // visibility | trust | conversion | engagement
  copyDirections: string;          // 1-2 sentences on what the copy should say/feel
}

export interface MessagingFramework {
  coreMessage: string;             // The one sentence the whole campaign communicates
  proofPoints: string[];           // 3 reasons the audience should believe it
  primaryCTA: string;              // The main action you want the audience to take
  secondaryCTA: string;            // A lower-commitment action
  toneKeywords: string[];          // 4-5 words that define the voice of this campaign
  thingsToAvoid: string[];         // 3 phrases/approaches that would dilute the message
}

export interface CampaignKPI {
  metric: string;                  // e.g. "LinkedIn connection acceptance rate"
  target: string;                  // e.g. ">40%"
  howToMeasure: string;            // e.g. "Check weekly in LinkedIn analytics"
  phase: number;                   // Which phase this KPI matters most in
}

export interface GeneratedCampaign {
  name: string;
  campaignType: string;
  coreMessage: string;
  targetAudience: string;
  audienceSegment: string;
  insights: string[];
  messagingFramework: MessagingFramework;
  phases: CampaignPhase[];
  channels: ChannelConfig[];
  contentPlan: ContentPiece[];
  kpis: CampaignKPI[];
  tasks: Array<{
    title: string;
    description: string;
    type: string;
    category: string;
    priority: number;
    estimatedDuration: number;
    taskEnergyType: string;
    phase: number;                 // Which phase this task belongs to
  }>;
}

function CollapsibleSection({ title, children, defaultOpen = true }: { title: string; children: React.ReactNode; defaultOpen?: boolean }) {
  const [open, setOpen] = useState(defaultOpen);
  return (
    <div>
      <button onClick={() => setOpen(!open)} className="flex items-center gap-2 w-full text-left mb-2">
        <p className="text-[10px] uppercase tracking-wider font-semibold text-slate-400 flex-1">{title}</p>
        {open ? <ChevronUp className="h-3 w-3 text-slate-400" /> : <ChevronDown className="h-3 w-3 text-slate-400" />}
      </button>
      {open && children}
    </div>
  );
}
